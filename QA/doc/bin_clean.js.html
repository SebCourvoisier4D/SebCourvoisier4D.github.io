#! /usr/bin/env node

/**
 * @file **(STABLE)** Clean all the folders used by the tests:
 * - The **Temp** folder: the folder used to download and extract the builds, as defined by the `tempDir` property of the configuration file
 * - The **Builds** folder: the folder used to store the builds, as defined by the `buildsDir` property of the configuration file
 * - The **Test** folder: the folder used to execute the tests, as defined by the `testDir` property of the configuration file
 * - The **Results** folder: the folder used to store the results of the tests, as defined by the `resultsDir` property of the configuration file
 *
 * Usage: `qa-clean [<buildsToKeep>]`
 * 
 * - `buildsToKeep` (`number`, optional): The number of builds to keep (for both the Results and Builds folders). It uses the `buildsToKeep` property of the configuration file by default.
 * 
 * Note: the alias `qa-clean` is made available using the `npm link` command.
 */

var qa = require('../'),
	main = new qa.QA(),
	config = new qa.Config(),
	cli_args = process.argv.slice(2),
	maxItems;

config.once('config not loaded', function(error) {
	console.log('Cannot load the configuration file:', error);
	process.exit(1);
});

config.once('config loaded', function() {
	if (global.qaConfig.buildsToKeep !== undefined) {
		maxItems = global.qaConfig.buildsToKeep;
	} else {
		maxItems = 5;
	}
	if (parseInt(cli_args[0]) > 0) {
		maxItems = parseInt(cli_args[0]);
	}
	main.cleanFolders(maxItems, function(err) {
		if (err) {
			console.log('Clean error :', err);
			process.exit(1);
		} else {
			console.log('Clean done');
			process.exit(0);
		}
	});
});

config.load();
