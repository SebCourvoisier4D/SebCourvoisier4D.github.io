#! /usr/bin/env node

/**
 * @file **(STABLE)** Sync the given test on the local P4 workspace.
 * 
 * Usage: `qa-sync-test <test_path> [--force]`
 * 
 * - `test_path` (`string`): Full path of the test to sync (path to its .json or .xml configuration file in the local P4 workspace).
 * - `--force` (optional): If set, performs a forced syncing.
 * 
 * Note: the alias `qa-sync-test` is made available using the `npm link` command.
 */

var qa = require('../'),
	config = new qa.Config(),
	synchronizer = new qa.Synchronizer(),
	cli_args = process.argv.slice(2);

config.once('config not loaded', function(error) {
	console.log('Cannot load the configuration file:', error);
	process.exit(1);
});

config.once('config loaded', function() {
	if (typeof cli_args[0] === 'string') {
		synchronizer.syncTest(cli_args[0], cli_args.indexOf('--force') !== -1);
	} else {
		console.log('Usage: qa-sync-test /path/to/test [--force]');
		process.exit(1);
	}
});

synchronizer.once('sync test failed', function(err, test) {
	console.log('Sync error for test "' + test + '":', err);
	process.exit(1);
});

synchronizer.once('sync test done', function(results, test) {
	console.log('Sync done for test "' + test + '":\n' + results.slice(-10).join('\n'));
	process.exit(0);
});

config.load();
