/**
 * @file **(STABLE)** This script is meant to be launched by the UnitTest extension of Wakanda Studio when running in test mode. It instantiates a WebSocket server that acts as a relay between the UnitTest extension and the {@link module:BuildRunner} module. It allows the test agent to send commands to the Studio and receive its events in real-time.
 */

 var async = require('async'),
	fs = require('fs-extra'),
	ws = require('nodejs-websocket'),
	server;

server = ws.createServer(function(conn) {
	sendMessageToStudio({
		status: 'wakbot connected'
	});
	conn.sendText('monitor connected');
	conn.on('text', function(message) {
		handleMessageFromWakbot(message.toString());
	});
	conn.once('close', function(code, reason) {
		sendMessageToStudio({
			status: 'wakbot disconnected'
		});
		process.exit(0);
	});
}).listen(8079);

sendMessageToStudio({
	status: 'monitor started'
});

process.stdin.setEncoding('utf8');
process.stdin.on('readable', function() {
	var message = process.stdin.read();
	if (message !== null) {
		handleMessageFromStudio(message.toString());
	}
});

function sendMessageToStudio(message) {
	if (message) {
		if (typeof message === 'object') {
			console.log(JSON.stringify(message));
		} else {
			console.log(message.toString());
		}
	}
}

function sendMessageToWakbot(message) {
	if (message && server) {
		server.connections.forEach(function(conn) {
			if (typeof message === 'object') {
				conn.sendText(JSON.stringify(message));
			} else {
				conn.sendText(message.toString());
			}
		});
	}
}

function handleMessageFromWakbot(message) {
	sendMessageToStudio(message);
}

function handleMessageFromStudio(message) {
	if (message === 'connect-to-server') {
		tryToConnectToServer(function (err, data) {
			if (err) {
				sendMessageToStudio('wserror:' + err);
			} else {
				sendMessageToStudio('ws:CONNECTED');
			}
		}, 0);
	}
	sendMessageToWakbot(message);
}

function tryToConnectToServer(cb, tries) {
    if (tries < 60) {
        var clientWS = ws.connect('ws://127.0.0.1:8080/remote_admin_ws/', {
            rejectUnauthorized: false
        });
        clientWS.once('text', function(data) {
            cb(null, data);
            clientWS.on('text', function(data) {
                if (/openingmode/i.test(data.toString()) === true && /defaultsolution/i.test(data.toString()) === false) {
                	sendMessageToStudio('ws:READY');
                }
            });
        });
        clientWS.once('error', function() {
            setTimeout(tryToConnectToServer, 500, cb, tries + 1);
        });
    } else {
        cb('wakanda server is unreachable', null);
    }
}
