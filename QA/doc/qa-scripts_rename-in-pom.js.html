/**
 * @file **(TO BE REMADE)** Perform a search/replace in all the `pom.xml` and `pom.json` files:
 *
 * `node /path/to/qa-scripts/rename-in-pom.js`
 * 
 * The search/replace patterns (as well as the P4 settings) have to be specified within the script itself (lines 18 to 26).
 */

 var glob = require('glob'),
	async = require('async'),
	fs = require('fs'),
	spawn = require('child_process').spawn,
	filesToModify = [],
	p4Checkout,
	params;

// ***** Change this: *****
params = {
	workspaceBasePath: 'C:/path/to/your/workspace',
	replacePattern: /ke-fong\.lin/gim,
	replaceWith: 'abdessamad.elkasimi',
	p4Server: 'srv-sources.private.4d.fr:1666',
	p4User: 'your.perforce.user.name',
	p4Pass: 'your.perforce.password',
	p4Workspace: 'your.perforce.workspace.name'
};
// ***** ***** ***** *****

glob(params.workspaceBasePath + '/depot/Wakanda/**/pom.*', function(err, files) {
	async.each(files, function(file, callback) {
		// Only do this on some branches:
		if (/wakanda.*main/i.test(file) || /wakanda.*wak10/i.test(file) || /wakanda.*wak11/i.test(file)) {
			fs.readFile(file, function(err, data) {
				if (!err && params.replacePattern.test(data.toString())) {
					filesToModify.push(file);
				}
				callback();
			});
		} else {
			callback();
		}
	}, function(err) {
		console.log(filesToModify.length + ' files to modify');
		async.each(filesToModify, function(file, callback) {
			p4Checkout = spawn('p4', ['-p' + params.p4Server, '-u' + params.p4User, '-P' + params.p4Pass, '-c' + params.p4Workspace, 'edit', file]);
			p4Checkout.on('error', function(err) {
				callback('error on ' + file + ': ' + err);
			});
			p4Checkout.on('exit', function(code, signal) {
				fs.readFile(file, function(err, data) {
					if (!err) {
						data = data.toString().replace(params.replacePattern, params.replaceWith);
						fs.writeFile(file, data, function(err) {
							callback();
						});
					} else {
						callback();
					}
				});
			});
			p4Checkout.stderr.on('data', function(err) {
				callback('error on ' + file + ': ' + err);
			});
		}, function(err) {
			console.log('done: ' + (err ? err : 'no error'));
		});
	});
});
